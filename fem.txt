SELECT uslugi.nazwa, uslugi.cena
FROM uslugi
where uslugi.cena >= 50;

-- 2
SELECT AVG(uslugi.cena) as "srednia cena uslug", COUNT(uslugi.id) as "liczba uslug"
from uslugi
where uslugi.rodzaj = 1;

on uslugi to 'kosmetyczka'@'localhost';
