1.SELECT film.ocena, count(*) as liczba_filmow
FROM film
GROUP by film.ocena

2.SELECT lista_filmow.kategoria, count(*) as liczba_filmow_w_kazdej_kategorii
FROM lista_filmow
JOIN kategoria on kategoria.kategoria_id = lista_filmow.FID
group by lista_filmow.kategoria;

3.SELECT film.tytul, count(*) as liczba_aktorow
FROM film
JOIN film_aktor on film_aktor.aktor_id = film.film_id
group by film.tytul;


4.SELECT klient.imie, klient.nazwisko, SUM(platnosc.kwota)
FROM platnosc
JOIN klient on klient.klient_id = platnosc.klient_id
group by klient.imie;

6.SELECT kategoria.nazwa, AVG(film.stawka_wypozycz), AVG(film.koszty_wymiany)
FROM kategoria
JOIN film_kategoria on film_kategoria.kategoria_id = kategoria.kategoria_id
join film on film.film_id = film_kategoria.film_id
group by kategoria.nazwa;


7.SELECT klient.imie, klient.nazwisko, AVG(platnosc.kwota)
FROM klient
JOIN platnosc on platnosc.klient_id = klient.klient_id
group by klient.imie;


8.SELECT pracownik.imie, pracownik.nazwisko, AVG(platnosc.kwota)
FROM pracownik
JOIN platnosc on platnosc.pracownik_id = pracownik.pracownik_id
group by pracownik.imie;


9.SELECT adres.adres, AVG(platnosc.kwota)
FROM adres
join sklep on sklep.adres_id = adres.adres_id
JOIN pracownik on pracownik.sklep_id = sklep.sklep_id
join platnosc on platnosc.pracownik_id = pracownik.pracownik_id
group by adres.adres;


10.SELECT kategoria.nazwa, AVG(platnosc.kwota), MAX(platnosc.kwota), MIN(platnosc.kwota), SUM(platnosc.kwota)
FROM platnosc
join wypozyczenie on wypozyczenie.wypozyczenie_id = platnosc.wypozyczenie_id
join spis on spis.spis_id = wypozyczenie.spis_id
join film on film.film_id = spis.film_id
join film_kategoria on film_kategoria.film_id = film.film_id
join kategoria on kategoria.kategoria_id = film_kategoria.kategoria_id
group by kategoria.nazwa;


11.SELECT klient.imie, klient.nazwisko, AVG(platnosc.kwota), MAX(platnosc.kwota), MIN(platnosc.kwota), SUM(platnosc.kwota)
FROM klient
join platnosc on platnosc.klient_id = klient.klient_id
group by klient.imie;


12.SELECT klient.imie, klient.nazwisko, wypozyczenie.data_wypozycz,  COUNT(spis.spis_id)
FROM klient
join wypozyczenie on wypozyczenie.klient_id = klient.klient_id
join spis on spis.spis_id = wypozyczenie.spis_id
group by klient.imie;


13.SELECT klient.imie, klient.nazwisko, wypozyczenie.data_wypozycz,  COUNT(platnosc.klient_id)
FROM klient
join wypozyczenie on wypozyczenie.klient_id = klient.klient_id
join platnosc on platnosc.wypozyczenie_id = wypozyczenie.wypozyczenie_id
group by klient.imie;
